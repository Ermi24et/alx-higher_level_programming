import module:
>>> MyList = __import__('1-my_list').MyList

Test positive unsorted integers:
>>> my_list = MyList()
>>> my_list.append(2)
>>> my_list.append(1)
>>> my_list.append(4)
>>> my_list.append(3)
>>> my_list.append(5)
>>> print(my_list)
[2, 1, 4, 3, 5]
>>> my_list.print_sorted()
[1, 2, 3, 4, 5]
>>> print(my_list)
[2, 1, 4, 3, 5]

Test for negative unsorted integers:
>>> my_list = MyList()
>>> my_list.append(-4)
>>> my_list.append(-6)
>>> my_list.append(-1)
>>> print(my_list)
[-4, -6, -1]
>>> my_list.print_sorted()
[-6, -4, -1]
>>> print(my_list)
[-4, -6, -1]
